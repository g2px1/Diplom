47template<class T>
48void ContainerPreparer<T>::init() {
49 // Now, we're checking available operations. We need to do this, to minimize delays
50 if (sequence::hasInsertFront(availableFunctions)) {
51 this->func_ptrs.emplace_back(
52 [](std::vector<int> &v, int value, int position = 0) { v.insert(v.begin(), value); });
53 }
54 if (sequence::hasInsertBack(availableFunctions)) {
55 this->func_ptrs.emplace_back(
56 [](std::vector<int> &v, int value, int position = 0) { v.insert(v.begin() + (v.size() - 1), value); });
57 }
58 if (sequence::hasInsertIndexed(availableFunctions)) {
59 this->func_ptrs.emplace_back(
60 [](std::vector<int> &v, int value, int position = 0) { v.insert(v.begin() + position, value); });
61 }
62 if (sequence::hasAccessElement(availableFunctions)) {
63 this->func_ptrs.emplace_back(
64 [](std::vector<int> &v, int value, int position = 0) { return v.at(position); });
65 }
66 if (sequence::hasEmplace(availableFunctions)) {
67 this->func_ptrs.emplace_back(
68 [](std::vector<int> &v, int value, int position = 0) { return v.emplace_back(value); });
69 }
70 if (sequence::hasDeleteFront(availableFunctions)) {
71 this->func_ptrs.emplace_back(
72 [](std::vector<int> &v, int value, int position = 0) { return v.erase(v.begin()); });
73 }
74 if (sequence::hasDeleteFront(availableFunctions)) {
75 this->func_ptrs.emplace_back(
76 [](std::vector<int> &v, int value, int position = 0) { return v.erase(v.begin() + (v.size() - 1)); });
77 }
78 if (sequence::hasDeleteIndexed(availableFunctions)) {
79 this->func_ptrs.emplace_back(
80 [](std::vector<int> &v, int value, int position = 0) { return v.erase(v.begin() + position); });
81 }
82}
